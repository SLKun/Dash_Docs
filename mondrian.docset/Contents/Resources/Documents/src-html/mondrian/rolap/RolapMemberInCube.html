010package mondrian.rolap;
011
012/**
013 * Extension to {@link RolapMember} that knows the current cube.
014 *
015 * <p>This is typical of members that occur in queries (where there is always a
016 * current cube) as opposed to members that belong to caches. Members of shared
017 * dimensions might occur in several different cubes, or even several times in
018 * the same virtual cube.
019 *
020 * @author jhyde
021 * @since 20 March, 2010
022 */
023public interface RolapMemberInCube extends RolapMember {
024    /**
025     * Returns the cube this cube member belongs to.
026     *
027     * <p>This method is not in the {@link RolapMember} interface, because
028     * regular members may be shared, and therefore do not belong to a specific
029     * cube.
030     *
031     * @return Cube this cube member belongs to, never null
032     */
033    RolapCube getCube();
034}
