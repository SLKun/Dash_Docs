029/**
030 * An axis within an OLAP {@link Query}.
031 *
032 * <p>An axis has a location (columns, rows, etc) and has zero or more
033 * dimensions that are placed on it.
034 *
035 * @author jdixon, Luc Boudreau
036 * @since May 29, 2007
037 */
038public class QueryAxis extends QueryNodeImpl {
039
040    protected final List<QueryDimension> dimensions = new DimensionList();
041
042    private final Query query;
043    protected Axis location = null;
044    private boolean nonEmpty;
045    private SortOrder sortOrder = null;
046    private String sortEvaluationLiteral = null;
047    private LimitFunction limitFunction = null;
048    private BigDecimal limitFunctionN = null;
049    private String limitFunctionSortLiteral = null;
050    private String filterCondition = null;
051    /**
052     * Creates a QueryAxis.
053     *
054     * @param query Query that the axis belongs to
055     * @param location Location of axis (e.g. ROWS, COLUMNS)
056     */
057    public QueryAxis(Query query, Axis location) {
058        super();
059        this.query = query;
060        this.location = location;
061    }
062
063    /**
064     * Returns the location of this <code>QueryAxis</code> in the query;
065     * <code>null</code> if unused.
066     *
067     * @return location of this axis in the query
068     */
069    public Axis getLocation() {
070        return location;
071    }
072
073    /**
074     * Returns a list of the dimensions placed on this QueryAxis.
075     *
076     * <p>Be aware that modifications to this list might
077     * have unpredictable consequences.</p>
078     *
079     * @return list of dimensions
080     */
081    public List<QueryDimension> getDimensions() {
082        return dimensions;
083    }
084
085    /**
086     * Returns the name of this QueryAxis.
087     *
088     * @return the name of this axis, for example "ROWS", "COLUMNS".
089     */
090    public String getName() {
091        return location.getCaption(query.getLocale());
092    }
093
094    /**
095     * Places a QueryDimension object one position before in the
096     * list of current dimensions. Uses a 0 based index.
097     * For example, to place the 5th dimension on the current axis
098     * one position before, one would need to call pullUp(4),
099     * so the dimension would then use axis index 4 and the previous
100     * dimension at that position gets pushed down one position.
101     * @param index The index of the dimension to move up one notch.
102     * It uses a zero based index.
103     */
104    public void pullUp(int index) {
105        Map<Integer, QueryNode> removed = new HashMap<Integer, QueryNode>();
106        removed.put(Integer.valueOf(index), this.dimensions.get(index));
107        Map<Integer, QueryNode> added = new HashMap<Integer, QueryNode>();
108        added.put(Integer.valueOf(index - 1), this.dimensions.get(index));
109        Collections.swap(this.dimensions, index, index - 1);
110        this.notifyRemove(removed);
111        this.notifyAdd(added);
112    }
113
114    /**
115     * Places a QueryDimension object one position lower in the
116     * list of current dimensions. Uses a 0 based index.
117     * For example, to place the 4th dimension on the current axis
118     * one position lower, one would need to call pushDown(3),
119     * so the dimension would then use axis index 4 and the previous
120     * dimension at that position gets pulled up one position.
121     * @param index The index of the dimension to move down one notch.
122     * It uses a zero based index.
123     */
124    public void pushDown(int index) {
125        Map<Integer, QueryNode> removed = new HashMap<Integer,  QueryNode>();
126        removed.put(Integer.valueOf(index), this.dimensions.get(index));
127        Map<Integer, QueryNode> added = new HashMap<Integer, QueryNode>();
128        added.put(Integer.valueOf(index + 1), this.dimensions.get(index));
129        Collections.swap(this.dimensions, index, index + 1);
130        this.notifyRemove(removed);
131        this.notifyAdd(added);
132    }
133
134    /**
135     * Places a {@link QueryDimension} object on this axis.
136     * @param dimension The {@link QueryDimension} object to add
137     * to this axis.
138     */
139    public void addDimension(QueryDimension dimension) {
140        this.getDimensions().add(dimension);
141        Integer index = Integer.valueOf(
142            this.getDimensions().indexOf(dimension));
143        this.notifyAdd(dimension, index);
144    }
145
146    /**
147     * Places a {@link QueryDimension} object on this axis at
148     * a specific index.
149     * @param dimension The {@link QueryDimension} object to add
150     * to this axis.
151     * @param index The position (0 based) onto which to place
152     * the QueryDimension
153     */
154    public void addDimension(int index, QueryDimension dimension) {
155        this.getDimensions().add(index, dimension);
156        this.notifyAdd(dimension, index);
157    }
158
159    /**
160     * Removes a {@link QueryDimension} object on this axis.
161     * @param dimension The {@link QueryDimension} object to remove
162     * from this axis.
163     */
164    public void removeDimension(QueryDimension dimension) {
165        Integer index = Integer.valueOf(
166            this.getDimensions().indexOf(dimension));
167        this.getDimensions().remove(dimension);
168        this.notifyRemove(dimension, index);
169    }

171    /**
172     * Returns whether this QueryAxis filters out empty rows.
173     * If true, axis filters out empty rows, and the MDX to evaluate the axis
174     * will be generated with the "NON EMPTY" expression.
175     *
176     * @return Whether this axis should filter out empty rows
177     *
178     * @see #setNonEmpty(boolean)
179     */
180    public boolean isNonEmpty() {
181        return nonEmpty;
182    }
183
184    /**
185     * Sets whether this QueryAxis filters out empty rows.
186     *
187     * @param nonEmpty Whether this axis should filter out empty rows
188     *
189     * @see #isNonEmpty()
190     */
191    public void setNonEmpty(boolean nonEmpty) {
192        this.nonEmpty = nonEmpty;
193    }
194
195    /**
196     * List of QueryDimension objects. The list is active: when a dimension
197     * is added to the list, it is removed from its previous axis.
198     */
199    private class DimensionList extends AbstractList<QueryDimension> {
200        private final List<QueryDimension> list =
201            new ArrayList<QueryDimension>();
202
203        public QueryDimension get(int index) {
204            return list.get(index);
205        }
206
207        public int size() {
208            return list.size();
209        }
210
211        public QueryDimension set(int index, QueryDimension dimension) {
212            if (dimension.getAxis() != null
213                && dimension.getAxis() != QueryAxis.this)
214            {
215                dimension.getAxis().getDimensions().remove(dimension);
216            }
217            dimension.setAxis(QueryAxis.this);
218            return list.set(index, dimension);
219        }
220
221        public void add(int index, QueryDimension dimension) {
222            if (this.contains(dimension)) {
223                throw new IllegalStateException(
224                    "dimension already on this axis");
225            }
226            if (dimension.getAxis() != null
227                && dimension.getAxis() != QueryAxis.this)
228            {
229                // careful! potential for loop
230                dimension.getAxis().getDimensions().remove(dimension);
231            }
232            dimension.setAxis(QueryAxis.this);
233            if (index >= list.size()) {
234                list.add(dimension);
235            } else {
236                list.add(index, dimension);
237            }
238        }
239
240        public QueryDimension remove(int index) {
241            QueryDimension dimension = list.remove(index);
242            dimension.setAxis(null);
243            return dimension;
244        }
245    }
246
247    void tearDown() {
248        for (QueryDimension node : this.getDimensions()) {
249            node.tearDown();
250        }
251        this.clearListeners();
252        this.getDimensions().clear();
253    }
254
255    /**
256     * <p>Sorts the axis according to the supplied order. The sort evaluation
257     * expression will be the default member of the default hierarchy of
258     * the dimension named "Measures".
259     * @param order The {@link SortOrder} to apply
260     * @throws OlapException If an error occurs while resolving
261     * the default measure of the underlying cube.
262     */
263    public void sort(SortOrder order) throws OlapException {
264        sort(
265            order,
266            query.getCube().getDimensions().get("Measures")
267                .getDefaultHierarchy().getDefaultMember());
268    }
269
270    /**
271     * Sorts the axis according to the supplied order
272     * and member unique name.
273     *
274     * <p>Using this method will try to resolve the supplied name
275     * parts from the underlying cube and find the corresponding
276     * member. This member will then be passed as a sort evaluation
277     * expression.
278     *
279     * @param order The {@link SortOrder} in which to
280     * sort the axis.
281     * @param nameParts The unique name parts of the sort
282     * evaluation expression.
283     * @throws OlapException If the supplied member cannot be resolved
284     *     with {@link org.olap4j.metadata.Cube#lookupMember(java.util.List)}
285     */
286    public void sort(SortOrder order, List<IdentifierSegment> nameParts)
287        throws OlapException
288    {
289        assert order != null;
290        assert nameParts != null;
291        Member member = query.getCube().lookupMember(nameParts);
292        if (member == null) {
293            throw new OlapException("Cannot find member.");
294        }
295        sort(order, member);
296    }
297
298    /**
299     * <p>Sorts the axis according to the supplied order
300     * and member.
301     * <p>This method is most commonly called by passing
302     * it a {@link Measure}.
303     * @param order The {@link SortOrder} in which to
304     * sort the axis.
305     * @param member The member that will be used as a sort
306     * evaluation expression.
307     */
308    public void sort(SortOrder order, Member member) {
309        assert order != null;
310        assert member != null;
311        sort(order, member.getUniqueName());
312    }
313
314    /**
315     * <p>Sorts the axis according to the supplied order
316     * and evaluation expression.
317     * <p>The string value passed as the sortEvaluationLitteral
318     * parameter will be used literally as a sort evaluator.
319     * @param order The {@link SortOrder} in which to
320     * sort the axis.
321     * @param sortEvaluationLiteral The literal expression that
322     * will be used to sort against.
323     */
324    public void sort(SortOrder order, String sortEvaluationLiteral) {
325        assert order != null;
326        assert sortEvaluationLiteral != null;
327        this.sortOrder = order;
328        this.sortEvaluationLiteral = sortEvaluationLiteral;
329    }
330
331    /**
332     * Clears the sort parameters from this axis.
333     */
334    public void clearSort() {
335        this.sortEvaluationLiteral = null;
336        this.sortOrder = null;
337    }
338
339    /**
340     * Returns the current sort order in which this
341     * axis will be sorted. Might return null of none
342     * is currently specified.
343     * @return The {@link SortOrder}
344     */
345    public SortOrder getSortOrder() {
346        return this.sortOrder;
347    }
348
349    /**
350     * Returns the current sort evaluation expression,
351     * or null if none are currently defined.
352     * @return The string literal that will be used in the
353     * MDX Order() function.
354     */
355    public String getSortIdentifierNodeName() {
356        return sortEvaluationLiteral;
357    }
358    
359    /**
360     * Returns a specified number of items from the 
361     *  top of the axis set.
362     * @param n
363     */
364    public void topCount(BigDecimal n) {
365        this.limitFunction = LimitFunction.TopCount;
366        this.limitFunctionN = n;
367    }
368
369    /**
370     * Returns a specified number of items from the 
371     * bottom of the axis set.
372     * @param n
373     */    
374    public void bottomCount(BigDecimal n) {
375        this.limitFunction = LimitFunction.BottomCount;
376        this.limitFunctionN = n;
377    }
378    
379    /**
380     * Limit the axis set to a specified number of items depending
381     * on the limitSortLiteral and {@link LimitFunction}.
382     * @param n - number of items/cumulative sum/percentage
383     */
384    public void limit(LimitFunction function, BigDecimal n, String limitSortLiteral) {
385        this.limitFunction = function;
386        this.limitFunctionN = n;
387        this.limitFunctionSortLiteral = limitSortLiteral;
388    }
389    
390    /**
391     * Clears the limit parameters of that axis.
392     */
393    public void clearLimitFunction() {
394        this.limitFunction = null;
395        this.limitFunctionN = null;
396        this.limitFunctionSortLiteral = null;
397    }
398
399        /**
400         * @return The {@link LimitFunction}
401         */
402        public LimitFunction getLimitFunction() {
403                return limitFunction;
404        }
405
406        /**
407         * @return The number of items or cumulative sum or percentage
408         * being used by the {@link LimitFunction}.
409         */
410        public BigDecimal getLimitFunctionN() {
411                return limitFunctionN;
412        }
413
414    /**
415     * Returns the current sort literal being used by the
416     * limit functionMight return null of none
417     * is currently specified.
418     * @return sort literal of the limit function
419     */ public String getLimitFunctionSortLiteral() {
420                return limitFunctionSortLiteral;
421        }
422
423     /**
424      * Filter the axis using the given condition before TopCount / Order
425      * are applied.
426      * @param filterCondition - the condition used for Filter()
427      */
428     public void filter(String filterCondition) {
429        this.filterCondition = filterCondition;
430     }
431     
432     /**
433      * Returns the Filter Condition used to filter the axis set.
434      * @return filter condition
435      */
436     public String getFilterCondition() {
437         return this.filterCondition;
438     }
439     
440     /**
441      * Clears the filter condition for this axis.
442      */
443     public void clearFilter() {
444         this.filterCondition = null;
445     }
