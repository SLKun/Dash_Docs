018package org.olap4j.driver.xmla.proxy;
019
020import org.olap4j.OlapException;
021import org.olap4j.driver.xmla.cache.XmlaOlap4jCache;
022
023import java.util.Map;
024
025/**
026 *
027 * Extended Proxy interface which supports cached SOAP calls.
028 *
029 * @author Luc Boudreau
030 */
031public interface XmlaOlap4jCachedProxy extends XmlaOlap4jProxy {
032
033    /**
034     * <p>Sets the cache class to use as a SOAP message cache.
035     *
036     * <p>Calling this method is not mandatory. If it isn't called,
037     * no cache will be used and all SOAP requests will be sent to
038     * the service end-point.
039     *
040     * @param configParameters This contains all the parameters used
041     * to configure the Olap4j driver. It contains the full class name
042     * of the cache implementation to use as well as the raw Cache
043     * config parameters.
044     * @param properties The properties to configure the cache,
045     * so all config parameters which started
046     * by Cache.* are inside this convenient thigny.
047     * @see XmlaOlap4jCache
048     */
049    void setCache(
050        Map<String, String> configParameters,
051        Map<String, String> properties) throws OlapException;
052
053}
054
055// End XmlaOlap4jCachedProxy.java
