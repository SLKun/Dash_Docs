018package org.olap4j.driver.xmla.proxy;
019
020import org.olap4j.driver.xmla.XmlaOlap4jDriver;
021import org.olap4j.driver.xmla.XmlaOlap4jServerInfos;
022import org.olap4j.impl.Base64;
023
024import java.io.*;
025import java.net.*;
026import java.util.concurrent.Future;
027import java.util.zip.GZIPInputStream;
028
029/**
030 * Extends the AbstractCachedProxy and serves as
031 * a production ready http communication class. Every SOAP request
032 * sends a POST call to the destination XMLA server and returns
033 * the response as a byte array, conforming to the Proxy interface.
034 *
035 * <p>It also takes advantage of the AbstractHttpProxy cookie
036 * managing facilities. All cookies received from the end point
037 * server will be sent back if they are not expired and they also
038 * conform to cookie domain rules.
039 *
040 * @author Luc Boudreau and Julian Hyde
041 */
042public class XmlaOlap4jHttpProxy extends XmlaOlap4jAbstractHttpProxy
043{
044    private final XmlaOlap4jDriver driver;
045
046    /**
047     * Creates a XmlaOlap4jHttpProxy.
048     *
049     * @param driver Driver
050     */
051    public XmlaOlap4jHttpProxy(
052        XmlaOlap4jDriver driver)
053    {
054        this.driver = driver;
055    }
056
057    private static final String DISCOVER =
058        "<Discover xmlns=\"urn:schemas-microsoft-com:xml-analysis\"";
059
060    private static final String EXECUTE =
061        "<Execute xmlns=\"urn:schemas-microsoft-com:xml-analysis\"";
062
063    @Override
064    public byte[] getResponse(XmlaOlap4jServerInfos serverInfos, String request)
065        throws XmlaOlap4jProxyException
066    {
067        URLConnection urlConnection = null;
068        try {
069            URL url = serverInfos.getUrl();
070            // Open connection to manipulate the properties
071            urlConnection = url.openConnection();
072            urlConnection.setDoOutput(true);
073
074            // Set headers
075            urlConnection.setRequestProperty(
076                "content-type",
077                "text/xml");
078            urlConnection.setRequestProperty(
079                "User-Agent",
080                "Olap4j("
081                    .concat(driver.getVersion())
082                    .concat(")"));
083            urlConnection.setRequestProperty(
084                "Accept",
085                "text/xml;q=1");
086            urlConnection.setRequestProperty(
087                "Accept-Charset",
088                getEncodingCharsetName()
089                    .concat(";q=1"));
090
091            // Tell the server that we support gzip encoding
092            urlConnection.setRequestProperty(
093                "Accept-Encoding",
094                "gzip");
095
096            // Some servers expect a SOAPAction header.
097            // TODO There is bound to be a better way to do this.
098            if (request.contains(DISCOVER)) {
099                urlConnection.setRequestProperty(
100                    "SOAPAction",
101                    "\"urn:schemas-microsoft-com:xml-analysis:Discover\"");
102            } else if (request.contains(EXECUTE)) {
103                urlConnection.setRequestProperty(
104                    "SOAPAction",
105                    "\"urn:schemas-microsoft-com:xml-analysis:Execute\"");
106            }
107
108            // Encode credentials for basic authentication
109            StringBuilder sb = new StringBuilder();
110            if (serverInfos.getUsername() != null
111                && serverInfos.getPassword() != null)
112            {
113                sb.append(serverInfos.getUsername());
114                sb.append(":");
115                sb.append(serverInfos.getPassword());
116            } else if (url.getUserInfo() != null) {
117                sb.append(url.getUserInfo());
118            }
119            if (!sb.toString().equals("")) {
120                String encoding =
121                    Base64.encodeBytes(
122                        sb.toString().getBytes(), 0);
123                urlConnection.setRequestProperty(
124                    "Authorization", "Basic " + encoding);
125            }
126
127            // Set correct cookies
128            this.useCookies(urlConnection);
129
130            // Send data (i.e. POST). Use same encoding as specified in the
131            // header.
132            final String encoding = getEncodingCharsetName();
133            urlConnection.getOutputStream().write(request.getBytes(encoding));
134
135            // Get the response, again assuming default encoding.
136            InputStream is = urlConnection.getInputStream();
137
138            // Detect that the server used gzip encoding
139            String contentEncoding =
140                urlConnection.getHeaderField("Content-Encoding");
141            if("gzip".equals(contentEncoding)) {
142                is = new GZIPInputStream(is);
143            }
144
145            final ByteArrayOutputStream baos = new ByteArrayOutputStream();
146            byte[] buf = new byte[1024];
147            int count;
148
149            while ((count = is.read(buf)) > 0) {
150                baos.write(buf, 0, count);
151            }
152
153            // Save the returned cookies for later use
154            this.saveCookies(urlConnection);
155
156            return baos.toByteArray();
157        // All exceptions should be trapped here.
158        // The response will only be available here anyways.
159        } catch (Exception e) {
160            // In order to prevent the JDK from keeping this connection
161            // in WAIT mode, we need to empty the error stream cache.
162            try {
163                final int espCode =
164                    ((HttpURLConnection)urlConnection).getResponseCode();
165                InputStream errorStream =
166                    ((HttpURLConnection)urlConnection).getErrorStream();
167                final ByteArrayOutputStream baos =
168                    new ByteArrayOutputStream();
169                final byte[] buf = new byte[1024];
170                int count;
171                if (errorStream != null) {
172                    while ((count = errorStream.read(buf)) > 0) {
173                        baos.write(buf, 0, count);
174                    }
175                    errorStream.close();
176                }
177                baos.close();
178            } catch (IOException ex) {
179                // Well, we tried. No point notifying the user here.
180            }
181            throw new XmlaOlap4jProxyException(
182                "This proxy encountered an exception while processing the "
183                + "query.",
184                e);
185        }
186    }
187
188    @Override
189    public Future<byte[]> getResponseViaSubmit(
190        final XmlaOlap4jServerInfos serverInfos,
191        final String request)
192    {
193        return XmlaOlap4jDriver.getFuture(this, serverInfos, request);
194    }
195
196    // implement XmlaOlap4jProxy
197    public String getEncodingCharsetName() {
198        return "UTF-8";
199    }
200}
