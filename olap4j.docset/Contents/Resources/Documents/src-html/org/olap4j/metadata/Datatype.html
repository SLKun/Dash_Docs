020/**
021 * Enumeration of the allowable data types of a Property or Measure.
022 *
023 * <p>The values derive from the OLE DB specification, specifically a
024 * subset of the OLE DB Types Indicators returned by SQL Server.
025 *
026 * @author jhyde
027 * @since Aug 23, 2006
028 */
029public enum Datatype implements XmlaConstant {
030    // The following values exactly match VARENUM
031    // in Automation and may be used in VARIANT.
032
033    INTEGER(3, "DBTYPE_I4", "A four-byte, signed integer: INTEGER"),
034
035    DOUBLE(5, "DBTYPE_R8", "A double-precision floating-point value: Double"),
036
037    CURRENCY(
038        6,
039        "DBTYPE_CY",
040        "A currency value: LARGE_INTEGER, Currency is a fixed-point number with "
041        + "four digits to the right of the decimal point. It is stored in an "
042        + "eight-byte signed integer, scaled by 10,000."),
043
044    BOOLEAN(
045        11,
046        "DBTYPE_BOOL",
047        "A Boolean value stored in the same way as in Automation: VARIANT_BOOL; "
048        + "0 means false and ~0 (bitwise, the value is not 0; that is, all bits "
049        + "are set to 1) means true."),
050
051    /**
052     * Used by SQL Server for value.
053     */
054    VARIANT(12, "DBTYPE_VARIANT", "An Automation VARIANT"),
055
056    /**
057     * Used by SQL Server for font size.
058     */
059    UNSIGNED_SHORT(18, "DBTYPE_UI2", "A two-byte, unsigned integer"),
060
061    /**
062     * Used by SQL Server for colors, font flags and cell ordinal.
063     */
064    UNSIGNED_INTEGER(19, "DBTYPE_UI4", "A four-byte, unsigned integer"),
065
066    // The following values exactly match VARENUM
067    // in Automation but cannot be used in VARIANT.
068
069    LARGE_INTEGER(
070        20,
071        "DBTYPE_I8",
072        "An eight-byte, signed integer: LARGE_INTEGER"),
073
074    // The following values are not in VARENUM in OLE.
075
076    STRING(
077        130,
078        "DBTYPE_WSTR",
079        "A null-terminated Unicode character string: wchar_t[length]; If "
080        + "DBTYPE_WSTR is used by itself, the number of bytes allocated "
081        + "for the string, including the null-termination character, is "
082        + "specified by cbMaxLen in the DBBINDING structure. If "
083        + "DBTYPE_WSTR is combined with DBTYPE_BYREF, the number of bytes "
084        + "allocated for the string, including the null-termination character, "
085        + "is at least the length of the string plus two. In either case, the "
086        + "actual length of the string is determined from the bound length "
087        + "value. The maximum length of the string is the number of allocated "
088        + "bytes divided by sizeof(wchar_t) and truncated to the nearest "
089        + "integer."),
090
091    /**
092     * Used by SAP BW. Represents a Character
093     */
094    ACCP(1000, "ACCP", "SAP BW Character"),
095
096    /**
097     * Used by SAP BW. Represents a CHAR
098     */
099    CHAR(1001, "CHAR", "SAP BW CHAR"),
100
101    /**
102     * Used by SAP BW. Represents a CHAR
103     */
104    CUKY(1002, "CUKY", "SAP BW CHAR"),
105
106    /**
107     * Used by SAP BW. Represents a Currency - Packed decimal, Integer
108     */
109    CURR(1003, "CURR", "SAP BW Currency - Packed decimal, Integer"),
110
111    /**
112     * Used by SAP BW. Represents a Date
113     */
114    DATS(1004, "DATS", "SAP BW Date"),
115
116    /**
117     * Used by SAP BW. Represents a Decimal
118     */
119    DEC(1005, "DEC", "SAP BW Decimal"),
120
121    /**
122     * Used by SAP BW. Represents a Point
123     */
124    FLTP(1006, "FLTP", "SAP BW Floating Point"),

128     */
129    INT1(1007, "INT1", "SAP BW Byte"),
130
131    /**
132     * Used by SAP BW. Represents a Small integer
133     */
134    INT2(1008, "INT2", "SAP BW Small integer"),
135
136    /**
137     * Used by SAP BW. Represents an Integer
138     */
139    INT4(1009, "INT4", "SAP BW Integer"),
140
141    /**
142     * Used by SAP BW. Represents a Text
143     */
144    LCHR(1010, "LCHR", "SAP BW Text"),
145
146    /**
147     * Used by SAP BW. Represents a Numeric
148     */
149    NUMC(1011, "NUMC", "SAP BW Numeric"),
150
151    /**
152     * Used by SAP BW. Represents a Tiny Int
153     */
154    PREC(1012, "PREC", "SAP BW Tiny Int"),
155
156    /**
157     * Used by SAP BW. Represents a QUAN Integer
158     */
159    QUAN(1013, "QUAN", "SAP BW QUAN Integer"),
160
161    /**
162     * Used by SAP BW. Represents a String
163     */
164    SSTR(1014, "SSTR", "SAP BW String"),

168     */
169    STRG(1015, "STRG", "SAP BW Long String"),
170
171    /**
172     * Used by SAP BW. Represents a Time
173     */
174    TIMS(1016, "TIMS", "SAP BW Time"),
175
176    /**
177     * Used by SAP BW. Represents a Varchar
178     */
179    VARC(1017, "VARC", "SAP BW Varchar"),
180
181    /**
182     * Used by SAP BW. Represents a Long String for Units
183     */
184    UNIT(1018, "UNIT", "SAP BW Long String for Units");
185
186    private final int xmlaOrdinal;
187    private String dbTypeIndicator;
188    private String description;
189
190    private static final DictionaryImpl<Datatype> DICTIONARY =
191        DictionaryImpl.forClass(Datatype.class);
192
193    Datatype(
194        int xmlaOrdinal,
195        String dbTypeIndicator,
196        String description)
197    {
198        this.xmlaOrdinal = xmlaOrdinal;
199        this.dbTypeIndicator = dbTypeIndicator;
200        this.description = description;
201    }
202
203    /**
204     * The internal name of this Datatype.
205     * Might not be unique across Datatype instances.
206     */
207    public String xmlaName() {
208        return dbTypeIndicator;
209    }
210
211    /**
212     * Human readable description of a Datatype instance.
213     */
214    public String getDescription() {
215        return description;
216    }
217
218    /**
219     * Unique identifier of a Datatype instance.
220     */
221    public int xmlaOrdinal() {
222        return xmlaOrdinal;
223    }
224
225    /**
226     * Per {@link org.olap4j.metadata.XmlaConstant}, returns a dictionary
227     * of all values of this enumeration.
228     *
229     * @return Dictionary of all values
230     */
231    public static Dictionary<Datatype> getDictionary() {
232        return DICTIONARY;
233    }
234}
